import React from "react";
// @material-ui/core components
import { makeStyles } from "@material-ui/core/styles";

// @material-ui/icons
import FastfoodIcon from "@material-ui/icons/Fastfood";
import AddAPhotoIcon from "@material-ui/icons/AddAPhoto";
// core components
import GridContainer from "/components/Grid/GridContainer.js";
import GridItem from "/components/Grid/GridItem.js";
import NavPills from "/components/NavPills/NavPills.js";

import styles from "/styles/jss/nextjs-material-kit/pages/componentsSections/pillsStyle.js";

const useStyles = makeStyles(styles);

/**
 * The function `SectionPills` renders a section with navigation pills for selecting between manual and
 * camera content.
 * @param props - The `props` parameter in the `SectionPills` function represents the properties that
 * are passed to the component when it is used. In this case, the `SectionPills` component expects two
 * props: `manualcontent` and `cameracontent`. These props are then used to render the content
 * @returns The `SectionPills` component is being returned. It contains a section with navigation pills
 * for "Manual" and "Camera" tabs, each with corresponding content passed as props (`ManualContent` and
 * `CameraContent`). The component structure includes a title "Selection" and icons for each tab
 * ("Fastfood" for Manual and "AddAPhoto" for Camera).
 */
export default function SectionPills(props) {
  const ManualContent = props.manualcontent;
  const CameraContent = props.cameracontent;
  const classes = useStyles();
  return (
    <div className={classes.section}>
      <div className={classes.container}>
        <div id="navigation-pills">
          <div className={classes.title}>
            <h3>Selection</h3>
          </div>
          <GridContainer>
            <GridItem xs={12} sm={12} md={16} lg={12}>
              <NavPills
                alignCenter
                color="primary"
                tabs={[
                  {
                    tabButton: "Manual",
                    tabIcon: FastfoodIcon,
                    tabContent: ManualContent,
                  },
                  {
                    tabButton: "Camera",
                    tabIcon: AddAPhotoIcon,
                    tabContent: CameraContent,
                  },
                ]}
              />
            </GridItem>
          </GridContainer>
        </div>
      </div>
    </div>
  );
}
